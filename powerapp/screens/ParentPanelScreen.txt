Set(gblParentTab, "approvals");

// Load pending approvals if online
If(Connection.Connected,
    ClearCollect(colPendingApprovals,
        AddColumns(
            Filter(DailyStatus,
                Status = "pending" &&
                ChildID = gblCurrentChild.ID),
            "ChoreName",
                LookUp(colChoresLocal, ID = ThisRecord.ChoreID).Title,
            "ChorePoints",
                LookUp(colChoresLocal, ID = ThisRecord.ChoreID).Points
        )
    ),
    // Offline: derive from local cache
    ClearCollect(colPendingApprovals,
        AddColumns(
            Filter(colDailyStatusLocal,
                Status = "pending" &&
                ChildID = gblCurrentChild.ID),
            "ChoreName",
                LookUp(colChoresLocal, ID = ThisRecord.ChoreID).Title,
            "ChorePoints",
                LookUp(colChoresLocal, ID = ThisRecord.ChoreID).Points
        )
    )
);

    Set(gblPINError, "Incorrect PIN. Try again.")

!IsBlank(gblPINError)

// lblPINError.Text:
gblPINError

// ════════════════════════════════════════════════════════════
// TAB: APPROVALS  (Visible = gblParentTab = "approvals")
// ════════════════════════════════════════════════════════════

// galApprovals.Items:
colPendingApprovals

// galApprovals item — lblApprovalChore.Text:
ThisItem.ChoreName

// galApprovals item — lblApprovalPoints.Text:
"+" & Text(ThisItem.ChorePoints) & " pts"

// galApprovals item — btnApprove.Text:  "✓ Approve"
// galApprovals item — btnApprove.Fill:  gblColors.Success
// galApprovals item — btnApprove.Color: gblColors.Card

// galApprovals item — btnApprove.OnSelect:
// Approve: update status → "approved", add points
With({rec: ThisItem},
    UpdateIf(colDailyStatusLocal,
        ChildID = rec.ChildID && ChoreID = rec.ChoreID && DateKey = rec.DateKey,
        {Status: "approved", IsCompleted: true});
    SaveData(colDailyStatusLocal, "cache_daily_status");

    Collect(colLedgerLocal, {
        FamilyID:    gblCurrentFamily.ID,
        ChildID:     rec.ChildID,
        DateKey:     gblTodayDate,
        EventType:   "chore_complete",
        RefID:       rec.ChoreID,
        PointsDelta: rec.ChorePoints,
        Note:        "Approved: " & rec.ChoreName
    });
    SaveData(colLedgerLocal, "cache_ledger");
    Set(gblPointsBalance, gblPointsBalance + rec.ChorePoints);

    If(Connection.Connected,
        Patch(DailyStatus,
            LookUp(DailyStatus,
                ChildID = rec.ChildID &&
                ChoreID = rec.ChoreID &&
                DateKey = rec.DateKey),
            {
                Status:      "approved",
                IsCompleted: true,
                VerifiedBy:  Office365Users.MyProfileV2().mail,
                VerifiedAt:  Now()
            });
        Patch(PointsLedger, Defaults(PointsLedger), {
            Title:       "Approved: " & rec.ChoreName,
            FamilyID:    gblCurrentFamily.ID,
            ChildID:     rec.ChildID,
            DateKey:     gblTodayDate,
            EventType:   "chore_complete",
            RefID:       rec.ChoreID,
            PointsDelta: rec.ChorePoints,
            Note:        "Approved by parent"
        })
    );

    RemoveIf(colPendingApprovals, ChildID = rec.ChildID && ChoreID = rec.ChoreID)
);

// galApprovals item — btnReject.Text:  "✗ Reject"
// galApprovals item — btnReject.Fill:  gblColors.Danger
// galApprovals item — btnReject.Color: gblColors.Card

// galApprovals item — btnReject.OnSelect:
With({rec: ThisItem},
    UpdateIf(colDailyStatusLocal,
        ChildID = rec.ChildID && ChoreID = rec.ChoreID && DateKey = rec.DateKey,
        {Status: "unchecked", IsCompleted: false});
    SaveData(colDailyStatusLocal, "cache_daily_status");

    If(Connection.Connected,
        Patch(DailyStatus,
            LookUp(DailyStatus,
                ChildID = rec.ChildID &&
                ChoreID = rec.ChoreID &&
                DateKey = rec.DateKey),
            {Status: "unchecked", IsCompleted: false})
    );

    RemoveIf(colPendingApprovals, ChildID = rec.ChildID && ChoreID = rec.ChoreID)
);

// lblNoApprovals.Visible:
CountRows(colPendingApprovals) = 0

// lblNoApprovals.Text:
"No pending approvals"

// ════════════════════════════════════════════════════════════

// ════════════════════════════════════════════════════════════
// Controls: sldBonusPoints (Slider), ddlBonusReason (Dropdown),
//           lblBonusAmt, btnAwardBonus, lblBonusSuccess

// sldBonusPoints: Min=1, Max=500, Default=25
// lblBonusAmt.Text:
"Award " & Text(sldBonusPoints.Value) & " points"

// ddlBonusReason.Items:
colBonusReasons

// btnAwardBonus.Text:
"Award Bonus"

// btnAwardBonus.Fill:
gblColors.Success

// btnAwardBonus.OnSelect:
Collect(colLedgerLocal, {
    FamilyID:    gblCurrentFamily.ID,
    ChildID:     gblCurrentChild.ID,
    DateKey:     gblTodayDate,
    EventType:   "bonus",
    PointsDelta: sldBonusPoints.Value,
    Note:        "Bonus: " & ddlBonusReason.SelectedText.Value
});
SaveData(colLedgerLocal, "cache_ledger");
Set(gblPointsBalance, gblPointsBalance + sldBonusPoints.Value);
Set(gblLifetimePoints, gblLifetimePoints + sldBonusPoints.Value);

Collect(colPendingSync, {
    Operation:   "POINTS_LEDGER",
    FamilyID:    gblCurrentFamily.ID,
    ChildID:     gblCurrentChild.ID,
    DateKey:     gblTodayDate,
    EventType:   "bonus",
    PointsDelta: sldBonusPoints.Value,
    Note:        "Bonus: " & ddlBonusReason.SelectedText.Value,
    Synced:      false
});
SaveData(colPendingSync, "cache_pending_sync");

If(Connection.Connected,
    Patch(PointsLedger, Defaults(PointsLedger), {
        Title:       "Bonus: " & ddlBonusReason.SelectedText.Value,
        FamilyID:    gblCurrentFamily.ID,
        ChildID:     gblCurrentChild.ID,
        DateKey:     gblTodayDate,
        EventType:   "bonus",
        PointsDelta: sldBonusPoints.Value,
        Note:        "Bonus: " & ddlBonusReason.SelectedText.Value
    })
);

Set(gblBonusSuccess, true);
UpdateContext({ctxBonusTimer: true});

//           btnDeductPoints, lblDeductConfirmMsg, lblDeductSuccess

// sldDeductPoints: Min=1, Max=500, Default=10

// lblDeductAmt.Text:
"Deduct " & Text(sldDeductPoints.Value) & " points"

// lblDeductAmt.Color:
gblColors.Danger

// ddlDeductReason.Items:
colDeductReasons

// btnDeductPoints.Text:
"Deduct Points"

// btnDeductPoints.Fill:
gblColors.Danger

// btnDeductPoints.Color:
gblColors.Card

// btnDeductPoints.OnSelect:
Collect(colLedgerLocal, {
    FamilyID:    gblCurrentFamily.ID,
    ChildID:     gblCurrentChild.ID,
    DateKey:     gblTodayDate,
    EventType:   "deduction",
    PointsDelta: -sldDeductPoints.Value,
    Note:        "Deduction: " & ddlDeductReason.SelectedText.Value
});
SaveData(colLedgerLocal, "cache_ledger");
Set(gblPointsBalance, gblPointsBalance - sldDeductPoints.Value);

    Operation:   "POINTS_LEDGER",
    FamilyID:    gblCurrentFamily.ID,
    ChildID:     gblCurrentChild.ID,
    DateKey:     gblTodayDate,
    EventType:   "deduction",
    PointsDelta: -sldDeductPoints.Value,
    Note:        "Deduction: " & ddlDeductReason.SelectedText.Value,
    Synced:      false
});
SaveData(colPendingSync, "cache_pending_sync");

If(Connection.Connected,
    Patch(PointsLedger, Defaults(PointsLedger), {
        Title:       "Deduction: " & ddlDeductReason.SelectedText.Value,
        FamilyID:    gblCurrentFamily.ID,
        ChildID:     gblCurrentChild.ID,
        DateKey:     gblTodayDate,
        EventType:   "deduction",
        PointsDelta: -sldDeductPoints.Value,
        Note:        "Deduction: " & ddlDeductReason.SelectedText.Value
    })
);

Set(gblDeductSuccess, true)

// ════════════════════════════════════════════════════════════
// TAB: SETTINGS  (Visible = gblParentTab = "settings")
// ════════════════════════════════════════════════════════════
// Controls: txtPrimaryEmail, txtSecondaryEmail, tglDailySummary,
//           txtSummaryTime, ddlTimezone, ddlApprovalMode,
//           sldApprovalThreshold, txtNewPortalPIN, btnSaveSettings

// btnSaveSettings.OnSelect:
If(Connection.Connected,
    Patch(FamilySettings,
        LookUp(FamilySettings, FamilyID = gblCurrentFamily.ID),
        {
            PrimaryEmail:          txtPrimaryEmail.Text,
            SecondaryEmail:        txtSecondaryEmail.Text,
            DailySummaryEnabled:   tglDailySummary.Value,
            SummaryTime:           txtSummaryTime.Text,
            Timezone:              ddlTimezone.SelectedText.Value,
            ApprovalMode:          ddlApprovalMode.SelectedText.Value,
            ApprovalThreshold:     sldApprovalThreshold.Value,
            ParentPortalPIN:       If(Len(txtNewPortalPIN.Text) >= 4,
                                       txtNewPortalPIN.Text,
                                       gblFamilySettings.ParentPortalPIN)
        }
    );
    Set(gblFamilySettings,
        LookUp(FamilySettings, FamilyID = gblCurrentFamily.ID));
    Set(gblSettingsSaved, true),

    Set(gblSettingsError, "You must be online to save settings.")
);

// lblSettingsSaved.Visible:  gblSettingsSaved
// lblSettingsSaved.Text:     "Settings saved!"
