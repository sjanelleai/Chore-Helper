// Chore Helper MVP
// Replit-friendly: no frameworks, simple state, localStorage persistence.

const STORAGE_KEY = "choreHelperState_v1";

const DEFAULT_STATE = {
  points: 0,
  chores: [
    { id: "make_bed", name: "Make bed", points: 10, completed: false },
    { id: "hamper", name: "Put clothes in hamper", points: 10, completed: false },
    { id: "pick_up_toys", name: "Pick up toys", points: 15, completed: false },
    { id: "feed_pet", name: "Feed pet", points: 15, completed: false },
    { id: "dishes_helper", name: "Dishes helper", points: 25, completed: false },
    { id: "clean_room_quick", name: "Clean room (quick)", points: 30, completed: false },
    { id: "homework_done", name: "Homework done", points: 50, completed: false },
    { id: "clean_room_deep", name: "Clean room (deep)", points: 100, completed: false }
  ],
  // Simple routine: reuse chores by ID
  routine: {
    morning: ["make_bed", "hamper"],
    afterSchool: ["homework_done", "pick_up_toys"],
    bedtime: ["clean_room_quick", "feed_pet"]
  },
  badges: [
    { id: "b50", name: "Starter Badge", threshold: 50, earned: false },
    { id: "b150", name: "Helper Level 2", threshold: 150, earned: false },
    { id: "b300", name: "Chore Master", threshold: 300, earned: false }
  ],
  store: [
    { id: "family_game", name: "Pick a family game", cost: 60 },
    { id: "treat", name: "Treat / Food", cost: 75 },
    { id: "build_house", name: "Build a house (creative time)", cost: 120 },
    { id: "arcade", name: "Arcade", cost: 200 }
  ],
  purchases: [] // { itemId, name, cost, ts }
};

let state = loadState();

// ---------- DOM ----------
const pointsTotalEl = document.getElementById("pointsTotal");
const pointsBigEl = document.getElementById("pointsBig");
const choresListEl = document.getElementById("choresList");
const badgesGridEl = document.getElementById("badgesGrid");
const storeGridEl = document.getElementById("storeGrid");
const purchaseListEl = document.getElementById("purchaseList");
const toastEl = document.getElementById("toast");

const viewChecklist = document.getElementById("view-checklist");
const viewTodo = document.getElementById("view-todo");
const viewPoints = document.getElementById("view-points");
const viewStore = document.getElementById("view-store");

const resetDailyBtn = document.getElementById("resetDailyBtn");

const routineMorningEl = document.getElementById("routineMorning");
const routineAfterSchoolEl = document.getElementById("routineAfterSchool");
const routineBedtimeEl = document.getElementById("routineBedtime");

const modalBackdrop = document.getElementById("modalBackdrop");
const modalTitle = document.getElementById("modalTitle");
const modalBody = document.getElementById("modalBody");
const modalCancel = document.getElementById("modalCancel");
const modalConfirm = document.getElementById("modalConfirm");

let modalAction = null;

// ---------- INIT ----------
wireNavButtons();
wireResetButton();
wireModalButtons();
render();

// ---------- Functions ----------
function loadState() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return structuredClone(DEFAULT_STATE);
    const parsed = JSON.parse(raw);

    // Basic merge to handle upgrades safely:
    // If something is missing, fall back to defaults.
    const merged = structuredClone(DEFAULT_STATE);
    merged.points = clampNumber(parsed.points, 0, 999999);

    merged.chores = Array.isArray(parsed.chores) ? parsed.chores : merged.chores;
    merged.routine = parsed.routine || merged.routine;
    merged.badges = Array.isArray(parsed.badges) ? parsed.badges : merged.badges;
    merged.store = Array.isArray(parsed.store) ? parsed.store : merged.store;
    merged.purchases = Array.isArray(parsed.purchases) ? parsed.purchases : merged.purchases;

    // Ensure required fields exist
    merged.chores = merged.chores.map(c => ({
      id: String(c.id),
      name: String(c.name),
      points: clampNumber(c.points, 0, 999999),
      completed: Boolean(c.completed)
    }));

    merged.badges = merged.badges.map(b => ({
      id: String(b.id),
      name: String(b.name),
      threshold: clampNumber(b.threshold, 0, 999999),
      earned: Boolean(b.earned)
    }));

    merged.store = merged.store.map(s => ({
      id: String(s.id),
      name: String(s.name),
      cost: clampNumber(s.cost, 0, 999999)
    }));

    merged.purchases = merged.purchases.map(p => ({
      itemId: String(p.itemId),
      name: String(p.name),
      cost: clampNumber(p.cost, 0, 999999),
      ts: clampNumber(p.ts, 0, 9999999999999)
    }));

    return merged;
  } catch (e) {
    console.warn("State load failed, using defaults.", e);
    return structuredClone(DEFAULT_STATE);
  }
}

function saveState() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
}

function clampNumber(n, min, max) {
  const x = Number(n);
  if (Number.isNaN(x)) return min;
  return Math.max(min, Math.min(max, x));
}

function render() {
  // Points
  pointsTotalEl.textContent = state.points;
  pointsBigEl.textContent = state.points;

  // Checklist
  choresListEl.innerHTML = "";
  state.chores.forEach(chore => {
    const row = document.createElement("div");
    row.className = "row" + (chore.completed ? " completed" : "");
    row.setAttribute("role", "button");
    row.setAttribute("tabindex", "0");
    row.title = "Tap to toggle";

    const left = document.createElement("div");
    left.className = "row-left";

    const check = document.createElement("div");
    check.className = "check";
    check.textContent = chore.completed ? "✅" : "⬜️";

    const name = document.createElement("div");
    name.className = "name";
    name.textContent = chore.name;

    left.appendChild(check);
    left.appendChild(name);

    const pts = document.createElement("div");
    pts.className = "points";
    pts.textContent = `+${chore.points}`;

    row.appendChild(left);
    row.appendChild(pts);

    row.addEventListener("click", () => toggleChore(chore.id));
    row.addEventListener("keydown", (e) => {
      if (e.key === "Enter" || e.key === " ") toggleChore(chore.id);
    });

    choresListEl.appendChild(row);
  });

  // Routine blocks (reuse chores by ID)
  renderRoutineBlock(routineMorningEl, state.routine.morning);
  renderRoutineBlock(routineAfterSchoolEl, state.routine.afterSchool);
  renderRoutineBlock(routineBedtimeEl, state.routine.bedtime);

  // Badges
  badgesGridEl.innerHTML = "";
  state.badges.forEach(badge => {
    const card = document.createElement("div");
    card.className = "badge" + (badge.earned ? " earned" : "");

    const top = document.createElement("div");
    top.className = "badge-top";

    const name = document.createElement("div");
    name.className = "badge-name";
    name.textContent = badge.earned ? `🏅 ${badge.name}` : `🔒 ${badge.name}`;

    const thr = document.createElement("div");
    thr.className = "badge-threshold";
    thr.textContent = `${badge.threshold} pts`;

    top.appendChild(name);
    top.appendChild(thr);

    const status = document.createElement("div");
    status.className = "badge-status";
    status.textContent = badge.earned ? "Earned!" : "Not yet";

    card.appendChild(top);
    card.appendChild(status);

    badgesGridEl.appendChild(card);
  });

  // Store
  storeGridEl.innerHTML = "";
  state.store.forEach(item => {
    const card = document.createElement("div");
    card.className = "store-item";

    const name = document.createElement("div");
    name.className = "store-name";
    name.textContent = item.name;

    const cost = document.createElement("div");
    cost.className = "store-cost";
    cost.textContent = `Cost: ${item.cost} points`;

    const actions = document.createElement("div");
    actions.className = "store-actions";

    const btn = document.createElement("button");
    btn.className = "btn btn-primary";
    btn.textContent = state.points >= item.cost ? "Buy" : "Need more points";
    if (state.points < item.cost) {
      btn.classList.add("btn-disabled");
      btn.disabled = true;
    } else {
      btn.addEventListener("click", () => confirmBuy(item.id));
    }

    actions.appendChild(btn);

    card.appendChild(name);
    card.appendChild(cost);
    card.appendChild(actions);

    storeGridEl.appendChild(card);
  });

  // Purchases
  purchaseListEl.innerHTML = "";
  if (state.purchases.length === 0) {
    const empty = document.createElement("div");
    empty.className = "row";
    empty.innerHTML = `<div class="row-left"><div class="check">🧾</div><div class="name">No purchases yet</div></div>
                       <div class="points">—</div>`;
    purchaseListEl.appendChild(empty);
  } else {
    // newest first
    [...state.purchases].reverse().forEach(p => {
      const row = document.createElement("div");
      row.className = "row";

      const left = document.createElement("div");
      left.className = "row-left";

      const icon = document.createElement("div");
      icon.className = "check";
      icon.textContent = "🛍️";

      const name = document.createElement("div");
      name.className = "name";
      const d = new Date(p.ts);
      name.textContent = `${p.name} (${d.toLocaleDateString()})`;

      left.appendChild(icon);
      left.appendChild(name);

      const cost = document.createElement("div");
      cost.className = "points";
      cost.textContent = `-${p.cost}`;

      row.appendChild(left);
      row.appendChild(cost);

      purchaseListEl.appendChild(row);
    });
  }
}

function renderRoutineBlock(containerEl, choreIds) {
  containerEl.innerHTML = "";

  choreIds.forEach(id => {
    const chore = state.chores.find(c => c.id === id);
    if (!chore) return;

    const row = document.createElement("div");
    row.className = "row" + (chore.completed ? " completed" : "");
    row.setAttribute("role", "button");
    row.setAttribute("tabindex", "0");

    const left = document.createElement("div");
    left.className = "row-left";

    const check = document.createElement("div");
    check.className = "check";
    check.textContent = chore.completed ? "✅" : "⬜️";

    const name = document.createElement("div");
    name.className = "name";
    name.textContent = chore.name;

    left.appendChild(check);
    left.appendChild(name);

    const pts = document.createElement("div");
    pts.className = "points";
    pts.textContent = `+${chore.points}`;

    row.appendChild(left);
    row.appendChild(pts);

    row.addEventListener("click", () => toggleChore(chore.id));
    row.addEventListener("keydown", (e) => {
      if (e.key === "Enter" || e.key === " ") toggleChore(chore.id);
    });

    containerEl.appendChild(row);
  });
}

function toggleChore(choreId) {
  const chore = state.chores.find(c => c.id === choreId);
  if (!chore) return;

  // Toggle
  chore.completed = !chore.completed;

  if (chore.completed) {
    state.points += chore.points;
    showToast(`+${chore.points} points! ✅`);
  } else {
    state.points -= chore.points;
    if (state.points < 0) state.points = 0;
    showToast(`-${chore.points} points (unchecked)`);
  }

  // Update badges
  checkBadges();

  saveState();
  render();
}

function checkBadges() {
  state.badges.forEach(b => {
    if (!b.earned && state.points >= b.threshold) {
      b.earned = true;
      showToast(`🏅 Badge earned: ${b.name}!`);
    }
  });
}

function confirmBuy(itemId) {
  const item = state.store.find(s => s.id === itemId);
  if (!item) return;

  openModal(
    "Confirm Purchase",
    `Spend ${item.cost} points on "${item.name}"?`,
    () => buyItem(itemId)
  );
}

function buyItem(itemId) {
  const item = state.store.find(s => s.id === itemId);
  if (!item) return;

  if (state.points < item.cost) {
    showToast("Not enough points!");
    return;
  }

  state.points -= item.cost;
  state.purchases.push({
    itemId: item.id,
    name: item.name,
    cost: item.cost,
    ts: Date.now()
  });

  saveState();
  render();
  showToast(`🛍️ Bought: ${item.name}!`);
}

function resetDaily() {
  // Reset chores only (keep points, badges, purchases)
  state.chores.forEach(c => (c.completed = false));
  saveState();
  render();
  showToast("Reset done. Tomorrow is a new mission! ↻");
}

function wireNavButtons() {
  const navButtons = document.querySelectorAll("[data-view]");
  navButtons.forEach(btn => {
    btn.addEventListener("click", () => {
      const view = btn.getAttribute("data-view");
      showView(view);
    });
  });
}

function showView(viewName) {
  // Hide all
  [viewChecklist, viewTodo, viewPoints, viewStore].forEach(v => v.classList.add("hidden"));

  // Show selected
  if (viewName === "checklist") viewChecklist.classList.remove("hidden");
  if (viewName === "todo") viewTodo.classList.remove("hidden");
  if (viewName === "points") viewPoints.classList.remove("hidden");
  if (viewName === "store") viewStore.classList.remove("hidden");

  // Scroll into view for nicer UX
  window.scrollTo({ top: 0, behavior: "smooth" });
}

function wireResetButton() {
  resetDailyBtn.addEventListener("click", () => {
    openModal(
      "Reset for Tomorrow",
      "This will uncheck all chores. Points and badges stay the same. Do it?",
      resetDaily
    );
  });
}

// Toast helper
let toastTimer = null;
function showToast(message) {
  toastEl.textContent = message;
  toastEl.classList.remove("hidden");
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => {
    toastEl.classList.add("hidden");
  }, 1800);
}

// Modal helpers
function wireModalButtons() {
  modalCancel.addEventListener("click", closeModal);
  modalBackdrop.addEventListener("click", (e) => {
    if (e.target === modalBackdrop) closeModal();
  });
  modalConfirm.addEventListener("click", () => {
    if (typeof modalAction === "function") modalAction();
    closeModal();
  });
}

function openModal(title, body, onConfirm) {
  modalTitle.textContent = title;
  modalBody.textContent = body;
  modalAction = onConfirm;
  modalBackdrop.classList.remove("hidden");
}

function closeModal() {
  modalAction = null;
  modalBackdrop.classList.add("hidden");
}
