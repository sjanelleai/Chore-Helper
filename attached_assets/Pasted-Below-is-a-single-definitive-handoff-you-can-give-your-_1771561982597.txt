This script:
	1.	Drops only the RPCs that commonly conflict (like add_child) without touching is_family_member (so you don’t have to CASCADE-drop policies).
	2.	Recreates helper + RPCs aligned to audited schema.
	3.	Creates Supabase-backed catalogs for chores and rewards (so they show again).
	4.	Adds the seeding RPC seed_default_catalog(family_id) you can call once per family (or auto-call from frontend when catalogs are empty).

-- 1) HELPER: current_family_id() — based on family_members
-- ============================================================
CREATE OR REPLACE FUNCTION public.current_family_id()
RETURNS uuid
LANGUAGE sql STABLE
SECURITY DEFINER
AS $$
  SELECT fm.family_id
  FROM public.family_members fm
  WHERE fm.user_id = auth.uid()
  LIMIT 1
$$;

GRANT EXECUTE ON FUNCTION public.current_family_id() TO authenticated;

-- ============================================================
-- 2) RPC: ensure_family_exists — create family + membership + settings
-- ============================================================
CREATE OR REPLACE FUNCTION public.ensure_family_exists(p_display_name text DEFAULT 'Parent')
RETURNS json
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_user_id uuid;
  v_family_id uuid;
  v_existing_family_id uuid;
  v_user_email text;
  v_family_name text;
BEGIN
  v_user_id := auth.uid();
  IF v_user_id IS NULL THEN
    RETURN json_build_object('error', 'Not authenticated');
  END IF;

  -- Fast path: already a member of a family
  SELECT fm.family_id INTO v_existing_family_id
  FROM public.family_members fm
  WHERE fm.user_id = v_user_id
  LIMIT 1;

  IF v_existing_family_id IS NOT NULL THEN
    SELECT f.name INTO v_family_name
    FROM public.families f
    WHERE f.id = v_existing_family_id;

    RETURN json_build_object(
      'family_id', v_existing_family_id,
      'display_name', COALESCE(v_family_name, p_display_name)
    );
  END IF;

  -- Slow path: create everything
  SELECT u.email INTO v_user_email
  FROM auth.users u
  WHERE u.id = v_user_id;

  INSERT INTO public.families (created_by, name)
  VALUES (v_user_id, COALESCE(p_display_name, 'My Family'))
  RETURNING id INTO v_family_id;

  INSERT INTO public.family_members (family_id, user_id, role)
  VALUES (v_family_id, v_user_id, 'parent')
  ON CONFLICT DO NOTHING;

  INSERT INTO public.family_settings (family_id, primary_parent_email)
  VALUES (v_family_id, v_user_email)
  ON CONFLICT (family_id) DO NOTHING;

  RETURN json_build_object(
    'family_id', v_family_id,
    'display_name', COALESCE(p_display_name, 'Parent')
  );
END;
$$;

GRANT EXECUTE ON FUNCTION public.ensure_family_exists(text) TO authenticated;

-- Chore catalog
CREATE TABLE IF NOT EXISTS public.chore_catalog (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  family_id uuid NOT NULL REFERENCES public.families(id) ON DELETE CASCADE,
  category text NOT NULL,
  title text NOT NULL,
  points int NOT NULL DEFAULT 10,
  sort_order int NOT NULL DEFAULT 0,
  active boolean NOT NULL DEFAULT true,
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now(),
  CONSTRAINT chore_catalog_unique UNIQUE (family_id, category, title)
);

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_trigger WHERE tgname = 'trg_chore_catalog_updated_at'
  ) THEN
    CREATE TRIGGER trg_chore_catalog_updated_at
    BEFORE UPDATE ON public.chore_catalog
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
  END IF;
END $$;

-- ============================================================
-- 5) RPC: add_child (uses children.display_name)
-- ============================================================
CREATE OR REPLACE FUNCTION public.add_child(p_display_name text)
RETURNS public.children
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_family_id uuid;
  v_child public.children;
BEGIN
  v_family_id := public.current_family_id();
  IF v_family_id IS NULL THEN
    RAISE EXCEPTION 'No family for current user';
  END IF;

  INSERT INTO public.children (family_id, display_name)
  VALUES (v_family_id, trim(p_display_name))
  RETURNING * INTO v_child;

  RETURN v_child;
END;
$$;

GRANT EXECUTE ON FUNCTION public.add_child(text) TO authenticated;

-- ============================================================
-- 6) RPC: seed_default_catalog(family_id)
--     Inserts chores/rewards only if empty for that family.
-- ============================================================
CREATE OR REPLACE FUNCTION public.seed_default_catalog(p_family_id uuid)
RETURNS json
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_has_chores boolean;
  v_has_rewards boolean;
  v_chores_added int := 0;
  v_rewards_added int := 0;
BEGIN
  -- Require membership (uses your existing policies/function assumptions)
  IF NOT EXISTS (
    SELECT 1 FROM public.family_members fm
    WHERE fm.family_id = p_family_id AND fm.user_id = auth.uid()
  ) THEN
    RETURN json_build_object('error', 'Not authorized');
  END IF;

  SELECT EXISTS (SELECT 1 FROM public.chore_catalog WHERE family_id = p_family_id) INTO v_has_chores;
  SELECT EXISTS (SELECT 1 FROM public.reward_catalog WHERE family_id = p_family_id) INTO v_has_rewards;

  IF NOT v_has_chores THEN
    INSERT INTO public.chore_catalog (family_id, category, title, points, sort_order)
    VALUES
      (p_family_id, 'Morning', 'Get dressed', 10, 10),
      (p_family_id, 'Morning', 'Brush teeth', 10, 20),
      (p_family_id, 'Morning', 'Make bed', 10, 30),
      (p_family_id, 'After School', 'Homework time', 15, 10),
      (p_family_id, 'After School', 'Put backpack away', 10, 20),
      (p_family_id, 'Evening', 'Clean up toys', 10, 10),
      (p_family_id, 'Evening', 'Shower / bath', 10, 20),
      (p_family_id, 'Evening', 'Read 15 minutes', 15, 30);
    GET DIAGNOSTICS v_chores_added = ROW_COUNT;
  END IF;

  IF NOT v_has_rewards THEN
    INSERT INTO public.reward_catalog (family_id, category, title, cost, requires_parent_approval, sort_order)
    VALUES
      (p_family_id, 'Small', 'Pick the family dessert', 50, true, 10),
      (p_family_id, 'Small', 'Extra 15 minutes screen time', 75, true, 20),
      (p_family_id, 'Medium', 'Choose a movie night', 150, true, 10),
      (p_family_id, 'Medium', 'Pick dinner spot (kid choice)', 200, true, 20),
      (p_family_id, 'Big', 'New toy (parent-approved)', 500, true, 10);
    GET DIAGNOSTICS v_rewards_added = ROW_COUNT;
  END IF;

  RETURN json_build_object(
    'seeded', (NOT v_has_chores) OR (NOT v_has_rewards),
    'chores_added', v_chores_added,
    'rewards_added', v_rewards_added
  );
END;
$$;

GRANT EXECUTE ON FUNCTION public.seed_default_catalog(uuid) TO authenticated;

-- ============================================================
-- 7) RPC: toggle_chore(child, chore, date) + ledger entry
-- ============================================================
CREATE OR REPLACE FUNCTION public.toggle_chore(p_child_id uuid, p_chore_id uuid, p_date_key date)
RETURNS json
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_family_id uuid;
  v_points int;
  v_completed boolean;
BEGIN
  -- Resolve family via child
  SELECT c.family_id INTO v_family_id
  FROM public.children c
  WHERE c.id = p_child_id;

  IF v_family_id IS NULL THEN
    RETURN json_build_object('error', 'Child not found');
  END IF;

  -- Require membership
  IF NOT EXISTS (
    SELECT 1 FROM public.family_members fm
    WHERE fm.family_id = v_family_id AND fm.user_id = auth.uid()
  ) THEN
    RETURN json_build_object('error', 'Not authorized');
  END IF;

  SELECT cc.points INTO v_points
  FROM public.chore_catalog cc
  WHERE cc.id = p_chore_id AND cc.family_id = v_family_id;

  IF v_points IS NULL THEN
    RETURN json_build_object('error', 'Chore not found');
  END IF;

  -- Upsert daily_status row and flip completion
  INSERT INTO public.daily_status (child_id, chore_id, date_key, completed, completed_at)
  VALUES (p_child_id, p_chore_id, p_date_key, true, now())
  ON CONFLICT (child_id, chore_id, date_key)
  DO UPDATE SET
    completed = NOT public.daily_status.completed,
    completed_at = CASE WHEN public.daily_status.completed = false THEN now() ELSE NULL END
  RETURNING completed INTO v_completed;

  -- Ledger delta
  INSERT INTO public.points_ledger (family_id, child_id, event_type, points_delta, ref_type, ref_id)
  VALUES (
    v_family_id,
    p_child_id,
    CASE WHEN v_completed THEN 'chore_complete' ELSE 'chore_uncomplete' END,
    CASE WHEN v_completed THEN v_points ELSE -v_points END,
    'chore',
    p_chore_id
  );

  RETURN json_build_object('ok', true, 'completed', v_completed);
END;
$$;

GRANT EXECUTE ON FUNCTION public.toggle_chore(uuid, uuid, date) TO authenticated;

-- ============================================================
-- 8) RPC: redeem_reward(child, reward) + ledger entry + redemption record
-- ============================================================
CREATE OR REPLACE FUNCTION public.redeem_reward(p_child_id uuid, p_reward_id uuid)
RETURNS json
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_family_id uuid;
  v_cost int;
  v_total int;
  v_redemption_id uuid;
BEGIN
  SELECT c.family_id INTO v_family_id
  FROM public.children c
  WHERE c.id = p_child_id;

  IF v_family_id IS NULL THEN
    RETURN json_build_object('error', 'Child not found');
  END IF;

  IF NOT EXISTS (
    SELECT 1 FROM public.family_members fm
    WHERE fm.family_id = v_family_id AND fm.user_id = auth.uid()
  ) THEN
    RETURN json_build_object('error', 'Not authorized');
  END IF;

  SELECT rc.cost INTO v_cost
  FROM public.reward_catalog rc
  WHERE rc.id = p_reward_id AND rc.family_id = v_family_id;

  IF v_cost IS NULL THEN
    RETURN json_build_object('error', 'Reward not found');
  END IF;

  SELECT COALESCE(SUM(pl.points_delta), 0) INTO v_total
  FROM public.points_ledger pl
  WHERE pl.child_id = p_child_id;

  IF v_total < v_cost THEN
    RETURN json_build_object('error', 'Insufficient points', 'total', v_total, 'cost', v_cost);
  END IF;

  INSERT INTO public.reward_redemptions (family_id, child_id, reward_id, cost, status)
  VALUES (v_family_id, p_child_id, p_reward_id, v_cost, 'redeemed')
  RETURNING id INTO v_redemption_id;

  INSERT INTO public.points_ledger (family_id, child_id, event_type, points_delta, ref_type, ref_id)
  VALUES (v_family_id, p_child_id, 'redeem', -v_cost, 'reward', p_reward_id);

  RETURN json_build_object('ok', true, 'redemption_id', v_redemption_id);
END;
$$;

GRANT EXECUTE ON FUNCTION public.redeem_reward(uuid, uuid) TO authenticated;

-- ============================================================
-- 9) RPC: grant_bonus(child, points, note)
-- ============================================================
CREATE OR REPLACE FUNCTION public.grant_bonus(p_child_id uuid, p_points int, p_note text)
RETURNS json
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_family_id uuid;
BEGIN
  SELECT c.family_id INTO v_family_id
  FROM public.children c
  WHERE c.id = p_child_id;

  IF v_family_id IS NULL THEN
    RETURN json_build_object('error', 'Child not found');
  END IF;

  IF NOT EXISTS (
    SELECT 1 FROM public.family_members fm
    WHERE fm.family_id = v_family_id AND fm.user_id = auth.uid()
      AND fm.role IN ('parent','guardian','admin')
  ) THEN
    RETURN json_build_object('error', 'Not authorized');
  END IF;

  INSERT INTO public.points_ledger (family_id, child_id, event_type, points_delta, note)
  VALUES (v_family_id, p_child_id, 'bonus', p_points, p_note);

  RETURN json_build_object('ok', true);
END;
$$;

GRANT EXECUTE ON FUNCTION public.grant_bonus(uuid, int, text) TO authenticated;

-- If your database already has is_family_member(uuid), use it.
-- If it does not, your existing policies would already be failing.
-- Policies below assume is_family_member(family_id) exists.

DO $$ BEGIN
  DROP POLICY IF EXISTS chore_catalog_select_member ON public.chore_catalog;
  DROP POLICY IF EXISTS reward_catalog_select_member ON public.reward_catalog;
  DROP POLICY IF EXISTS daily_status_select_member ON public.daily_status;
  DROP POLICY IF EXISTS points_ledger_select_member ON public.points_ledger;
  DROP POLICY IF EXISTS reward_redemptions_select_member ON public.reward_redemptions;
EXCEPTION WHEN undefined_object THEN
  -- ignore if not present
END $$;

CREATE POLICY chore_catalog_select_member
  ON public.chore_catalog
  FOR SELECT
  USING (is_family_member(family_id));

CREATE POLICY reward_catalog_select_member
  ON public.reward_catalog
  FOR SELECT
  USING (is_family_member(family_id));

CREATE POLICY points_ledger_select_member
  ON public.points_ledger
  FOR SELECT
  USING (is_family_member(family_id));

CREATE POLICY reward_redemptions_select_member
  ON public.reward_redemptions
  FOR SELECT
  USING (is_family_member(family_id));

-- daily_status requires family membership via child->family join
CREATE POLICY daily_status_select_member
  ON public.daily_status
  FOR SELECT
  USING (
    EXISTS (
      SELECT 1
      FROM public.children c
      WHERE c.id = daily_status.child_id
        AND is_family_member(c.family_id)
    )
  );

What your developer must change in the frontend (to make chores/rewards show up)

1) Catalog reads

Replace any “frontend constants” catalog usage with Supabase reads:
	•	chore_catalog filtered by family_id and active=true
	•	reward_catalog filtered by family_id and active=true

2) Catalog initialization (why nothing is showing)

On first load after login:
	•	call ensure_family_exists() → get family_id
	•	fetch chore_catalog + reward_catalog for that family_id
	•	if empty, call seed_default_catalog(family_id)
	•	refetch catalogs and render

3) Chore toggles (fix the error you showed)

Any “toggle” logic must write to:
	•	rpc('toggle_chore', { p_child_id, p_chore_id, p_date_key })
and the UI must read completion from:
	•	daily_status joined with chore_catalog for that day

4) Points totals

Stop depending on any legacy points table.
Points total = sum(points_ledger.points_delta) per child.

⸻

Do you need to run a query in Supabase “to connect databases”?

No. Supabase doesn’t require any “connection query.” The only required Supabase-side action is running the SQL migration above (tables + RPCs). Your app connects via the existing SUPABASE_URL + ANON_KEY env vars.
